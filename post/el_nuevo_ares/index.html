#!/usr/bin/env bash
# =============================================================================
# torrent_streamer.sh — "El nuevo Ares"
# Basado en: https://imlauera.github.io/post/el_nuevo_ares/
#
# Uso:
#   ./torrent_streamer.sh "Jason vs Freddy"   → busca y reproduce
#   ./torrent_streamer.sh --reset             → mata todo y arranca de cero
#   ./torrent_streamer.sh --setup             → instala dependencias + configura
#   ./torrent_streamer.sh --help
# =============================================================================

set -euo pipefail

# ─── Colores ─────────────────────────────────────────────────────────────────
RED='\033[0;31m'; YELLOW='\033[1;33m'; GREEN='\033[0;32m'
CYAN='\033[0;36m'; MAGENTA='\033[0;35m'; BOLD='\033[1m'; RESET='\033[0m'

info()  { echo -e "${CYAN}[INFO]${RESET}  $*"; }
ok()    { echo -e "${GREEN}[OK]${RESET}    $*"; }
warn()  { echo -e "${YELLOW}[WARN]${RESET}  $*"; }
step()  { echo -e "\n${MAGENTA}[STEP]${RESET} ${BOLD}$*${RESET}"; }
error() { echo -e "${RED}[ERROR]${RESET} $*" >&2; }
die()   { error "$*"; exit 1; }

# ─── Configuración ───────────────────────────────────────────────────────────
QBT_HOST="http://127.0.0.1:8080"
QBT_USER="admin"
QBT_PASS="adminadmin"
QBT_COOKIES="/tmp/qbt_cookies.txt"
QBT_CONF="$HOME/.config/qBittorrent/qBittorrent.conf"
QBT_LOG="/tmp/qbt.log"

JACKETT_HOST="http://127.0.0.1:9117"
JACKETT_ENGINES_DIR="$HOME/.local/share/qBittorrent/nova3/engines"
JACKETT_KEY_FILE="$JACKETT_ENGINES_DIR/jackett.json"
JACKETT_PLUGIN_URL="https://raw.githubusercontent.com/qbittorrent/search-plugins/master/nova3/engines/jackett.py"

DOWNLOAD_DIR="$HOME/Downloads/torrents"

# Indexers que requieren FlareSolverr (mencionados en el artículo)
FLARESOLVERR_INDEXERS=("0magnet" "1337x" "52bt")

# ─── Parseo de argumentos ────────────────────────────────────────────────────
MODE="stream"
QUERY="Jason vs Freddy"

usage() {
    echo -e "${BOLD}Uso:${RESET}"
    echo "  $0 \"nombre del torrent\"   Buscar y reproducir"
    echo "  $0 --setup                Instalar dependencias y configurar"
    echo "  $0 --reset                Reiniciar qBittorrent y Jackett desde cero"
    echo "  $0 --help                 Esta ayuda"
    exit 0
}

for arg in "$@"; do
    case "$arg" in
        --help|-h) usage ;;
        --reset)   MODE="reset" ;;
        --setup)   MODE="setup" ;;
        *)         QUERY="$arg" ;;
    esac
done

# ─── Banner ──────────────────────────────────────────────────────────────────
banner() {
    echo -e "\n${BOLD}${CYAN}╔══════════════════════════════════════════════╗${RESET}"
    echo -e "${BOLD}${CYAN}║   🎬  El Nuevo Ares — Torrent Streamer       ║${RESET}"
    echo -e "${BOLD}${CYAN}║   qBittorrent-nox + Jackett + mpv            ║${RESET}"
    echo -e "${BOLD}${CYAN}╚══════════════════════════════════════════════╝${RESET}"
    echo -e "   ${CYAN}https://imlauera.github.io/post/el_nuevo_ares/${RESET}\n"
}

# ═══════════════════════════════════════════════════════════════════════════════
# MODO: --reset  (cuando "se pone estúpido")
# ═══════════════════════════════════════════════════════════════════════════════
do_reset() {
    step "Reiniciando todo desde cero..."

    info "Matando qBittorrent-nox..."
    pkill -f qbittorrent-nox 2>/dev/null && ok "qBittorrent-nox detenido." \
        || info "No había proceso activo."
    rm -f "$QBT_COOKIES" "$QBT_LOG"

    info "Reiniciando Jackett..."
    sudo systemctl restart jackett 2>/dev/null && ok "Jackett reiniciado." \
        || warn "No se pudo reiniciar Jackett con systemd."

    info "Limpiando temporales..."
    rm -f /tmp/qbt_*.txt /tmp/qbt*.log

    ok "Reset completo. Corré el script nuevamente."
    exit 0
}

# ═══════════════════════════════════════════════════════════════════════════════
# MODO: --setup
# ═══════════════════════════════════════════════════════════════════════════════
do_setup() {
    step "Setup inicial — instalando dependencias..."

    if command -v apt &>/dev/null; then
        info "Distro Debian/Ubuntu detectada."
        sudo apt update -qq
        sudo apt install -y qbittorrent-nox jackett curl jq mpv python3 \
            && ok "Paquetes instalados."
    elif command -v yay &>/dev/null; then
        info "Arch Linux con yay detectado."
        # Según el artículo: yay -S jackett-bin para Arch
        yay -S --noconfirm qbittorrent-nox jackett-bin curl jq mpv \
            && ok "Paquetes instalados."
    elif command -v pacman &>/dev/null; then
        sudo pacman -Sy --noconfirm qbittorrent-nox curl jq mpv
        warn "Instalá jackett desde AUR: yay -S jackett-bin"
    else
        warn "Distro no reconocida. Instalá manualmente: qbittorrent-nox jackett curl jq mpv"
    fi

    step "Configurando jackett.json..."
    setup_jackett_json

    step "Descargando plugin de Jackett para qBittorrent..."
    mkdir -p "$JACKETT_ENGINES_DIR"
    curl -sL "$JACKETT_PLUGIN_URL" -o "$JACKETT_ENGINES_DIR/jackett.py" \
        && ok "Plugin descargado: $JACKETT_ENGINES_DIR/jackett.py" \
        || warn "No se pudo descargar. URL manual: $JACKETT_PLUGIN_URL"

    step "Configurando qBittorrent.conf..."
    configure_qbt

    echo ""
    ok "Setup completo."
    echo -e "\n  ${BOLD}Próximos pasos:${RESET}"
    echo -e "  1. Abrí Jackett en ${CYAN}http://localhost:9117${RESET}"
    echo -e "  2. Copiá tu API key y pegala en: ${CYAN}$JACKETT_KEY_FILE${RESET}"
    echo -e "  3. Agregá los indexers públicos en Jackett (checkbox por checkbox)"
    echo -e "  4. Iniciá systemd: ${BOLD}sudo systemctl start jackett${RESET}"
    echo -e "  5. Corré: ${BOLD}./torrent_streamer.sh \"nombre pelicula\"${RESET}"
    echo ""
    warn "Indexers que necesitan FlareSolverr (no funcionan sin él):"
    for idx in "${FLARESOLVERR_INDEXERS[@]}"; do
        echo -e "     ${RED}✗${RESET} $idx"
    done
    echo ""
    exit 0
}

# ═══════════════════════════════════════════════════════════════════════════════
# FUNCIONES COMUNES
# ═══════════════════════════════════════════════════════════════════════════════

check_deps() {
    step "Verificando dependencias..."
    local missing=0

    for cmd in qbittorrent-nox curl jq mpv python3; do
        if ! command -v "$cmd" &>/dev/null; then
            warn "  ✗ '$cmd' no está instalado."
            missing=1
        else
            ok "  ✓ $cmd"
        fi
    done

    if ! command -v jackett &>/dev/null && \
       ! systemctl list-units --type=service 2>/dev/null | grep -q jackett; then
        warn "  ✗ jackett no encontrado. En Arch: yay -S jackett-bin | En Debian: apt install jackett"
        missing=1
    else
        ok "  ✓ jackett"
    fi

    [[ $missing -eq 1 ]] && die "Faltan dependencias. Corré: $0 --setup"
}

# Crea jackett.json con todos los campos del artículo si no existe
setup_jackett_json() {
    mkdir -p "$JACKETT_ENGINES_DIR"

    if [[ ! -f "$JACKETT_KEY_FILE" ]]; then
        info "Creando $JACKETT_KEY_FILE..."
        cat > "$JACKETT_KEY_FILE" <<'EOF'
{
    "api_key": "YOUR_API_KEY_HERE",
    "url": "http://127.0.0.1:9117",
    "tracker_first": false,
    "thread_count": 20
}
EOF
        warn "Editá $JACKETT_KEY_FILE con tu API key de http://localhost:9117"
    else
        # Completar campos faltantes según el artículo
        local updated
        updated=$(jq '
          . + {
            "url":          (.url          // "http://127.0.0.1:9117"),
            "tracker_first":(.tracker_first // false),
            "thread_count": (.thread_count  // 20)
          }' "$JACKETT_KEY_FILE")
        echo "$updated" > "$JACKETT_KEY_FILE"
        ok "jackett.json verificado y completo."
    fi
}

configure_qbt() {
    info "Configurando qBittorrent.conf (password: adminadmin)..."
    mkdir -p "$(dirname "$QBT_CONF")"

    local PASS_LINE
    PASS_LINE='WebUI\Password_PBKDF2="@ByteArray(ARQ77eY1NUZaQsuDHbIMCA==:0WMRkYTUWVT9wVvdDtHAjU9b3b7uB8NR1Gur2hmQCvCDpm39Q+PsJRJPaCU51dEiz+dTzh8qbPsL8WkFljQYFQ==)"'

    if [[ ! -f "$QBT_CONF" ]]; then
        cat > "$QBT_CONF" <<EOF
[Preferences]
${PASS_LINE}
WebUI\Port=8080
WebUI\Username=admin
WebUI\LocalHostAuth=false
EOF
        ok "qBittorrent.conf creado."
    elif grep -qF 'WebUI\Password_PBKDF2' "$QBT_CONF"; then
        ok "Contraseña ya configurada."
    else
        sed -i "/^\[Preferences\]/a ${PASS_LINE}" "$QBT_CONF"
        ok "Contraseña insertada en qBittorrent.conf."
    fi
}

start_qbt() {
    if curl -sf "$QBT_HOST/api/v2/app/version" &>/dev/null; then
        ok "qBittorrent-nox ya está corriendo."
        return
    fi

    info "Iniciando qBittorrent-nox en segundo plano..."
    mkdir -p "$DOWNLOAD_DIR"
    nohup qbittorrent-nox \
        --webui-port=8080 \
        --save-path="$DOWNLOAD_DIR" \
        >"$QBT_LOG" 2>&1 &
    local pid=$!

    local tries=0
    while ! curl -sf "$QBT_HOST/api/v2/app/version" &>/dev/null; do
        sleep 1
        (( tries++ ))
        if [[ $tries -ge 20 ]]; then
            error "No levantó. Últimas líneas del log:"
            tail -10 "$QBT_LOG" >&2
            die "Revisá $QBT_LOG o corré: $0 --reset"
        fi
    done
    ok "qBittorrent-nox corriendo (PID $pid)."
}

login_qbt() {
    info "Login en qBittorrent WebUI..."
    local result
    result=$(curl -sf -X POST "$QBT_HOST/api/v2/auth/login" \
        -d "username=${QBT_USER}&password=${QBT_PASS}" \
        -c "$QBT_COOKIES" || echo "")

    case "$result" in
        "Ok.")    ok "Login exitoso." ;;
        "Fails.") die "Login fallido. Corré --reset y volvé a intentar." ;;
        *)        warn "Respuesta login: '${result:-vacía}' (puede ser sesión activa)." ;;
    esac
}

start_jackett() {
    info "Iniciando Jackett (systemd)..."

    if systemctl is-active --quiet jackett 2>/dev/null; then
        ok "Jackett ya está activo."
    else
        sudo systemctl start jackett 2>/dev/null \
            && ok "Jackett iniciado." \
            || warn "No se pudo iniciar con systemd. ¿Instalado? (yay -S jackett-bin)"
    fi

    local tries=0
    while ! curl -sf "$JACKETT_HOST" &>/dev/null; do
        sleep 1
        (( tries++ ))
        [[ $tries -ge 25 ]] && die "Jackett no responde en $JACKETT_HOST"
    done
    ok "Jackett disponible."
}

get_jackett_key() {
    setup_jackett_json

    JACKETT_API_KEY=$(jq -r '.api_key // empty' "$JACKETT_KEY_FILE" 2>/dev/null || echo "")

    if [[ -z "$JACKETT_API_KEY" || "$JACKETT_API_KEY" == "YOUR_API_KEY_HERE" ]]; then
        die "API key no configurada. Editá $JACKETT_KEY_FILE con la key de http://localhost:9117"
    fi
    ok "API key lista."
}

warn_flaresolverr() {
    echo ""
    warn "Nota: estos indexers requieren FlareSolverr para funcionar:"
    for idx in "${FLARESOLVERR_INDEXERS[@]}"; do
        echo -e "     ${YELLOW}⚠${RESET}  $idx"
    done
    warn "Si no obtenés resultados, configurá FlareSolverr en Jackett."
    echo ""
}

search_torrent() {
    local query_encoded
    query_encoded=$(python3 -c \
        "import urllib.parse,sys; print(urllib.parse.quote(sys.argv[1]))" "$QUERY")

    info "Buscando '${BOLD}${QUERY}${RESET}' en todos los indexers..."

    SEARCH_RESULTS=$(curl -sf \
        "${JACKETT_HOST}/api/v2.0/indexers/all/results?Query=${query_encoded}&apikey=${JACKETT_API_KEY}") \
        || die "Error consultando Jackett."

    local total
    total=$(echo "$SEARCH_RESULTS" | jq '.Results | length' 2>/dev/null || echo 0)

    if [[ "$total" -eq 0 ]]; then
        warn_flaresolverr
        die "Sin resultados para '${QUERY}'."
    fi

    info "Resultados: ${BOLD}$total${RESET}"
    echo ""
    info "Top 5 por seeds:"
    echo "$SEARCH_RESULTS" | jq -r '
      .Results
      | map(select(.Seeders > 0))
      | sort_by(-.Seeders)
      | .[0:5]
      | .[]
      | "  \(.Seeders) seeds | \(.Title[:65])"
    ' 2>/dev/null || true
    echo ""
}

pick_torrent() {
    info "Seleccionando: menor calidad disponible con más seeds..."

    # 1. Intento: sin 4K/1080p/Remux
    MAGNET=$(echo "$SEARCH_RESULTS" | jq -r '
      .Results
      | map(select(
          .Seeders > 0 and
          (.Guid | startswith("magnet:?")) and
          (.Title | test("4K|2160p|UHD|1080p|FHD|REMUX"; "i") | not)
        ))
      | sort_by(-.Seeders)
      | .[0].Guid // empty
    ')

    # 2. Fallback: sin 4K al menos
    if [[ -z "$MAGNET" || "$MAGNET" == "null" ]]; then
        warn "No hay baja calidad. Probando con 1080p..."
        MAGNET=$(echo "$SEARCH_RESULTS" | jq -r '
          .Results
          | map(select(
              .Seeders > 0 and
              (.Guid | startswith("magnet:?")) and
              (.Title | test("4K|2160p|UHD"; "i") | not)
            ))
          | sort_by(-.Seeders)
          | .[0].Guid // empty
        ')
    fi

    # 3. Fallback final: cualquier magnet
    if [[ -z "$MAGNET" || "$MAGNET" == "null" ]]; then
        warn "Usando cualquier calidad disponible..."
        MAGNET=$(echo "$SEARCH_RESULTS" | jq -r '
          .Results
          | map(select(.Seeders > 0 and (.Guid | startswith("magnet:?"))))
          | sort_by(-.Seeders)
          | .[0].Guid // empty
        ')
    fi

    if [[ -z "$MAGNET" || "$MAGNET" == "null" ]]; then
        warn_flaresolverr
        die "No se encontró magnet link. Revisá los indexers activos en Jackett."
    fi

    local title seeders
    title=$(echo "$SEARCH_RESULTS" | jq -r --arg g "$MAGNET" \
        '.Results | map(select(.Guid == $g)) | .[0].Title // "desconocido"')
    seeders=$(echo "$SEARCH_RESULTS" | jq -r --arg g "$MAGNET" \
        '.Results | map(select(.Guid == $g)) | .[0].Seeders // 0')

    ok "Elegido: ${BOLD}${title}${RESET}"
    ok "Seeds:   ${BOLD}${seeders}${RESET}"
    info "Magnet:  ${MAGNET:0:90}..."
}

add_torrent() {
    info "Agregando a qBittorrent-nox..."

    local resp
    resp=$(curl -sf -X POST "$QBT_HOST/api/v2/torrents/add" \
        -b "$QBT_COOKIES" \
        --data-urlencode "urls=$MAGNET" \
        -d "savepath=$DOWNLOAD_DIR" \
        -d "category=streamed" \
        -d "sequentialDownload=true" \
        -d "firstLastPiecePrio=true")

    ok "Respuesta qBittorrent: ${resp:-Ok.}"
    info "Guardando en: $DOWNLOAD_DIR"
}

wait_and_play() {
    info "Esperando datos suficientes para reproducir... (Ctrl+C para cancelar)"

    local found_file="" tries=0 max_wait=360 min_size=52428800

    while [[ $tries -lt $max_wait ]]; do
        found_file=$(find "$DOWNLOAD_DIR" -type f \
            \( -iname "*.mkv" -o -iname "*.mp4" -o -iname "*.avi" \
               -o -iname "*.mov" -o -iname "*.webm" -o -iname "*.ts" \
               -o -iname "*.m4v" -o -iname "*.m2ts" \) \
            -newer "$QBT_COOKIES" 2>/dev/null | head -1)

        if [[ -n "$found_file" ]]; then
            local size size_mb
            size=$(stat -c%s "$found_file" 2>/dev/null || echo 0)
            size_mb=$(( size / 1048576 ))
            [[ $size -gt $min_size ]] && { ok "Listo (${size_mb} MB)"; break; }
            info "Descargando... ${size_mb} MB / 50 MB mínimo"
        else
            info "Esperando inicio de descarga... (${tries}s)"
        fi

        sleep 5
        (( tries += 5 ))
    done

    if [[ -z "$found_file" ]]; then
        warn "No apareció archivo de video en $DOWNLOAD_DIR tras ${max_wait}s."
        warn "Abrilo manualmente: mpv \"$DOWNLOAD_DIR/<archivo>\""
        exit 0
    fi

    echo ""
    ok "▶  Reproduciendo: $(basename "$found_file")"
    mpv \
        --cache=yes \
        --cache-secs=120 \
        --demuxer-readahead-secs=60 \
        --force-seekable=yes \
        --title="🎬 $(basename "$found_file")" \
        "$found_file"
}

# ═══════════════════════════════════════════════════════════════════════════════
# MAIN
# ═══════════════════════════════════════════════════════════════════════════════
banner

case "$MODE" in
    reset)  do_reset ;;
    setup)  do_setup ;;
    stream)
        info "Búsqueda: '${BOLD}${QUERY}${RESET}'"
        echo ""
        check_deps
        configure_qbt
        start_qbt
        login_qbt
        start_jackett
        get_jackett_key
        search_torrent
        pick_torrent
        add_torrent
        wait_and_play
        echo ""
        ok "¡Disfrutá la película! 🍿"
        ;;
esac
