SDL_Surface *surface_from_file(const char *file_path){
    int w, h, n;
    unsigned char *pixels = stbi_load(file_path, &w, &h, &n, STBI_rgb_alpha);
    if (pixels == NULL){
        fprintf(stderr, "ERROR: could not load file %s\n", file_path, stbi_failure_reason());
        exit(1);
    }
}
