Instalación Alpine Linux — Mini Root FS Bootstrap

Particionado (UEFI/GPT)

fdisk -l  # identificá el disco destino, asumimos /dev/sdb
cfdisk /dev/sdb  # seleccioná gpt
# EFI: 300M, Swap: 4G, Root: resto

Con fdisk si preferís:

fdisk /dev/sdb
g
n; 1; 2048; +300M; t; 1; 1   # EFI
n; 2; default; +4G; t; 2; swap
n; 3; default; default
w

Formateo y montado

mkfs.ext4 /dev/sdb3
mkswap /dev/sdb2
mkfs.fat -F 32 /dev/sdb1
mount /dev/sdb3 /mnt
mkdir -p /mnt/boot/efi
mount /dev/sdb1 /mnt/boot/efi
swapon /dev/sdb2

Bootstrap: descargar y extraer el tarball

wget https://dl-cdn.alpinelinux.org/alpine/v3.21/releases/x86_64/alpine-minirootfs-3.21.3-x86_64.tar.gz

# el flag -p es OBLIGATORIO, sin el los permisos quedan mal
tar -xzpf alpine-minirootfs-3.21.3-x86_64.tar.gz -C /mnt

Versión más reciente siempre en: https://alpinelinux.org/downloads/

Preparar el chroot

mount -t proc none /mnt/proc
mount -t sysfs none /mnt/sys
mount --bind /dev /mnt/dev
mount --bind /dev/pts /mnt/dev/pts
cp /etc/resolv.conf /mnt/etc/resolv.conf
chroot /mnt /bin/sh

Si tenés arch-chroot disponible en el host, reemplaza todo lo anterior con:

arch-chroot /mnt

Dentro del chroot

# Repositorios
cat > /etc/apk/repositories << EOF
https://dl-cdn.alpinelinux.org/alpine/v3.21/main
https://dl-cdn.alpinelinux.org/alpine/v3.21/community
EOF

apk update
apk add alpine-base linux-lts linux-firmware

# Timezone
apk add tzdata
cp /usr/share/zoneinfo/America/Argentina/Buenos_Aires /etc/localtime
echo "America/Argentina/Buenos_Aires" > /etc/timezone

# Hostname
echo "alpine" > /etc/hostname
cat > /etc/hosts << EOF
127.0.0.1   localhost
::1         localhost
127.0.1.1   alpine.localdomain  alpine
EOF

# Teclado español
apk add kbd-bkeymaps
echo "KEYMAP=es" > /etc/conf.d/keymaps
rc-update add keymaps default

# Password root
passwd

# fstab — mirá los UUID con blkid y completá
blkid
cat > /etc/fstab << EOF
UUID=<uuid-sdb3>  /          ext4  defaults,noatime  0 1
UUID=<uuid-sdb1>  /boot/efi  vfat  defaults          0 2
UUID=<uuid-sdb2>  none       swap  sw                0 0
EOF

# Servicios base (OpenRC, no systemd)
rc-update add devfs sysinit
rc-update add dmesg sysinit
rc-update add mdev sysinit
rc-update add hwclock boot
rc-update add modules boot
rc-update add sysctl boot
rc-update add hostname boot
rc-update add bootmisc boot
rc-update add syslog boot
rc-update add mount-ro shutdown
rc-update add killprocs shutdown
rc-update add savecache shutdown

GRUB (UEFI)

apk add grub grub-efi efibootmgr
grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=alpine
grub-mkconfig -o /boot/grub/grub.cfg

Si el firmware tiene bugs y borra la entrada EFI: grub-install --target=x86_64-efi --efi-directory=/boot/efi --removable

BIOS/MBR legacy:

apk add grub
grub-install /dev/sdb
grub-mkconfig -o /boot/grub/grub.cfg

Salir y reiniciar

exit
umount -R /mnt
reboot

Post-instalación (ya en Alpine)

# Usuario
adduser -G users,video,audio tu_usuario
addgroup tu_usuario wheel
apk add sudo
echo "%wheel ALL=(ALL:ALL) ALL" > /etc/sudoers.d/wheel

# Shell
apk add bash fish
chsh -s /bin/bash tu_usuario

# Wifi
apk add networkmanager networkmanager-wifi wpa_supplicant
rc-update add networkmanager default
rc-service networkmanager start
nmtui  # para conectarse

                        ## OpenRC vs systemd — equivalencias

                        | systemd (Arch)              | OpenRC (Alpine)                    |
                        |-----------------------------|------------------------------------|
                        | `systemctl enable X`        | `rc-update add X default`          |
                        | `systemctl start X`         | `rc-service X start`               |
                        | `systemctl disable X`       | `rc-update del X default`          |
                        | `systemctl status X`        | `rc-service X status`              |
                        | `systemctl restart X`       | `rc-service X restart`             |

                        ---

                        ## apk vs pacman — equivalencias

                        | pacman (Arch)    | apk (Alpine)                  |
                        |------------------|-------------------------------|
                        | `pacman -Syu`    | `apk upgrade`                 |
                        | `pacman -S pkg`  | `apk add pkg`                 |
                        | `pacman -Rcs`    | `apk del pkg`                 |
                        | `pacman -Ss`     | `apk search`                  |
                        | `pacman -Si`     | `apk info pkg`                |
                        | `pacman -Qe`     | `apk list --installed`        |
                        | `pacman -Ql`     | `apk info --contents pkg`     |
                        | `pacman -Scc`    | `apk cache clean`             |
