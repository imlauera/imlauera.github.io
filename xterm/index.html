XTerm.vt100.scrollBar:false
XTerm.vt100.bellIsUrgent:true

XTerm.vt100.faceName:Liberation Mono
XTerm.vt100.faceSize:12

    Ctrl Shift <Key>Prior: larger-vt-font() \n\
    Ctrl Shift <Key>Next: smaller-vt-font()
